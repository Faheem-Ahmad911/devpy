# DevPy Website - Nginx Configuration for Redirects and SEO
# Add this to your server block in nginx.conf

server {
    listen 80;
    server_name devpy.tech www.devpy.tech;
    return 301 https://www.devpy.tech$request_uri;
}

server {
    listen 443 ssl http2;
    server_name devpy.tech;
    return 301 https://www.devpy.tech$request_uri;
}

server {
    listen 443 ssl http2;
    server_name www.devpy.tech;
    
    # Your SSL configuration here
    
    root /path/to/your/website/public;
    index index.html;
    
    # Remove trailing slashes
    rewrite ^/(.*)/$ /$1 permanent;
    
    # Redirect index.html to root
    if ($request_uri ~ "^/index\.html$") {
        return 301 https://www.devpy.tech/;
    }
    
    # Custom error pages
    error_page 404 /404.html;
    
    # Security headers
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    
    # Cache static assets
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|pdf|woff|woff2)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
    
    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript image/svg+xml;
}
